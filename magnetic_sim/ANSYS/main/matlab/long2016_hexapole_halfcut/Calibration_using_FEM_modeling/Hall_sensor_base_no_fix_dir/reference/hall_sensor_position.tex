{\large\textbf{霍爾感測器安裝位置}}

\medskip
單位向量 $\mathbf{e}_1$：
\[
\mathbf{e}_1 =
\begin{bmatrix}
\cos\psi\cos\theta \\[2pt]
\cos\psi\sin\theta \\[2pt]
\sin\psi
\end{bmatrix}
\]

\vspace{1em}

{\large\textbf{極尖往霍爾感測器之單位向量}}

\medskip
(1)\quad
\[
\mathbf{e}_2 =
\begin{bmatrix}
\cos(\psi+\beta)\cos\theta \\[2pt]
\cos(\psi+\beta)\sin\theta \\[2pt]
\sin(\psi+\beta)
\end{bmatrix}
\]

\[
\mathbf{e}_3 =
\begin{bmatrix}
\cos\!\big[90^\circ+(\psi+\beta)\big]\cos\theta \\[2pt]
\cos\!\big[90^\circ+(\psi+\beta)\big]\sin\theta \\[2pt]
\sin\!\big[90^\circ+(\psi+\beta)\big]
\end{bmatrix}
\]

\vspace{1em}

{\large\textbf{因此最終安裝位置}} $\Rightarrow$
\[
\begin{bmatrix} P_1 \\ P_2 \\ P_3 \end{bmatrix}
= \hat{\ell}\,\mathbf{e}_1 + 4.572\,\mathbf{e}_2 + 0.41\,\mathbf{e}_3 \quad (\mathrm{mm})
\]

其中：
\begin{itemize}[leftmargin=2.5em,topsep=2pt,itemsep=1pt]
  \item $\hat{\ell} = $ 工作空間半徑
  \item $\psi = $ 仰角
  \item $\beta = $ 磁極半錐角
  \item $\theta = $ 方位角
\end{itemize}
