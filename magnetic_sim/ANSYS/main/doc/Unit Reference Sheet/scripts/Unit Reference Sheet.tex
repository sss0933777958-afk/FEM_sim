\textbf{Current Base Model}
\[
b_{ij} = \underbrace{\left(\tfrac{6}{5}k_{11}\right)\left(\frac{N_c}{4\pi\,\widehat{\ell}^{\,2}\,R_a}\right)}_{\textstyle {}^{B}\widehat{g}_{I}}\; S_i\,\overline{K}_{I}\,I
= S_i\Big(\underbrace{{}^{B}\widehat{g}_{I}\,\overline{K}_{I}\,I}_{\textstyle G}\Big)
\]

\vspace{10pt}

\textbf{Voltage Base Model}
\[
b_{ij} = S_i\cdot D\cdot V
= S_i\cdot\Big[\underbrace{\left(\tfrac{6\,d_{11}}{5}\right)}_{\textstyle {}^{B}\widehat{g}_{V}}\overline{D}\,V\Big]
= S_i\Big(\underbrace{{}^{B}\widehat{g}_{V}\,\overline{D}\,V}_{\textstyle G}\Big)
\]
