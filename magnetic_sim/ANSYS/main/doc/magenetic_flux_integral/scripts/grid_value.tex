\section*{2. 決定一個網格的量值}

\subsection*{(1) 取座標 $(y, z) = (0, -13.01)$ 的節點說明}

\subsection*{(2) 將問題簡化成 2D,計算整個三角形面積,定義向量:}

\[
\begin{cases}
\vec{V}_1 = P_2 - P_1 = (-3.51 \times 10^{-3},\; -7.77 \times 10^{-3}) \\[4pt]
\vec{V}_2 = P_3 - P_1 = (-7 \times 10^{-5},\; -8.02 \times 10^{-3})
\end{cases}
\]

整個三角形面積:
\[
A_{tot} = \frac{1}{2}\left|\vec{V}_1 \times \vec{V}_2\right| = 1.38 \times 10^{-5}\,\mathrm{mm}^2
\]

子三角形面積與權重:
\[
\begin{aligned}
A_1 &= (Q, P_2, P_3) = 1.177 \times 10^{-5}\,\mathrm{mm}^2 \\
A_2 &= (Q, P_1, P_3) = 0.0255 \times 10^{-5}\,\mathrm{mm}^2 \\
A_3 &= (Q, P_2, P_1) = 0.17661 \times 10^{-5}\,\mathrm{mm}^2
\end{aligned}
\quad\Rightarrow\quad
\begin{cases}
W_1 = \dfrac{A_1}{A_{tot}} = \dfrac{1.177 \times 10^{-5}}{1.38 \times 10^{-5}} = 0.853 \\[8pt]
W_2 = \dfrac{A_2}{A_{tot}} = 0.0185 \\[8pt]
W_3 = \dfrac{A_3}{A_{tot}} = 0.128
\end{cases}
\]

\subsection*{(3) 由重心插值得磁通密度與磁通量}

因此
\[
B(0, -13.01) = B_{x_1} \cdot W_1 + B_{x_2} \cdot W_2 + B_{x_3} \cdot W_3 = 257.7\,\mathrm{mT}
\]

故
\[
\Delta \Phi = B(0, -13.01) \cdot \Delta A = 257.7 \times 10^{-3} \cdot 0.5 \times 10^{-12} = 128.85\,\mathrm{fWb}
\]

\subsection*{(4) 實際使用 3D 插值}

\[
B(0, -13.01) = 362.733\,\mathrm{mT}
\]

故
\[
\Delta \Phi = 362.733\,\mathrm{mT} \cdot 0.5\,\mu\mathrm{m}^2 = 181.4\,\mathrm{fWb}
\]

全部加總:
\[
\boxed{\;\Phi = \sum \Delta \Phi = 0.572\,\mathrm{nWb}\;}
\]
